#-------------------------Build Stage---------------------------------
FROM node:22-alpine3.19 AS builder

# Set the working directory
WORKDIR /app

# Install Infisical CLI
RUN apk add --no-cache bash curl && curl -1sLf \
  'https://dl.cloudsmith.io/public/infisical/infisical-cli/setup.alpine.sh' | bash \
  && apk add infisical

# Arguments for Infisical (passed during build)
ARG INFISICAL_API_URL
ARG INFISICAL_TOKEN
ARG ENV
ARG PROJECT_ID

# Copy package files and install dependencies
COPY package*.json ./
RUN npm install

# Copy the entire project
COPY . .

# Export secrets to .env file during build using Machine Identity
RUN infisical export --token=$INFISICAL_TOKEN --domain=$INFISICAL_API_URL --projectId=${PROJECT_ID} --env=${ENV} --format=dotenv > .env

#-------------------------Final Stage---------------------------------
    
FROM node:22-alpine3.19

# Set working directory
WORKDIR /app

# Copy production artifacts from builder
COPY --from=builder /app/package*.json ./
COPY --from=builder /app/node_modules ./node_modules
COPY --from=builder /app/index.js ./
COPY --from=builder /app/.env* ./

# Use non-root user for security
USER node

# Start the application
CMD ["npm", "start"]
